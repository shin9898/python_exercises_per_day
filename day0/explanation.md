def maximum_of_the_product(n: int, arr: list[int]) -> int:
    max_prod = arr[0]
    current_max = arr[0]
    current_min = arr[0]

    for i in range(1, n):

        if num < 0:
            current_max, current_min = current_min, current_max
        current_max = max(num, current_max * num)
        current_min = min(num, current_min * num)
        max_prod = max(max_prod, current_max)

    return max_prod
